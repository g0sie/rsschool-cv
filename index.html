def ips_between(start: str, end: str) -> int:
    '''
    For two IPv4 addresses returns number of addresses between them
    (including the first one, excluding the last one)
    '''
    start_lst = list(map(lambda x: int(x), start.split('.')))
    end_lst = list(map(lambda x: int(x), end.split('.')))
    res = sum([2 ** (8 * (3 - i)) * (end_lst[i] - start_lst[i]) for i in range(4)])
    return res
